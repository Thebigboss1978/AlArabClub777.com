11. سكربت Node.js لقراءة .env

🟢 لتشغيل الـAgent محليًا مع المفاتيح بشكل تلقائي:

// server.js
import express from "express";
import dotenv from "dotenv";
import fetch from "node-fetch";

dotenv.config({ path: "./env/.env" });

const app = express();
const PORT = process.env.PORT || 3000;

// Simple AI Agent endpoint with error handling + rate limiting placeholder
app.get("/ai", async (req, res) => {
  try {
    const q = req.query.q || "مرحبا";
    const response = await fetch("https://api.openai.com/v1/chat/completions", {
      method: "POST",
      headers: {
        "Content-Type": "application/json",
        "Authorization": `Bearer ${process.env.OPENAI_API_KEY}`
      },
      body: JSON.stringify({
        model: "gpt-4o-mini",
        messages: [{ role: "user", content: q }]
      })
    });
    const data = await response.json();
    res.json(data);
  } catch (err) {
    res.status(500).json({ error: "❌ فشل الاتصال مع AI Endpoint" });
  }
});

app.listen(PORT, () => console.log(`✅ Server running on http://localhost:${PORT}`));

➡️ قم بتشغيل npm install express dotenv node-fetch ثم node server.js. يتم تحميل المفاتيح من .env ويصبح الـAgent جاهز. (اختياري: أضف rate limiting / auth بسيط).

// frontend example
async function askAgent(q){
  const res = await fetch(`/ai?q=${encodeURIComponent(q)}`);
  const data = await res.json();
  console.log(data);
  document.getElementById("aiBox").textContent = data.choices?.[0]?.message?.content || "لا يوجد رد";
}
